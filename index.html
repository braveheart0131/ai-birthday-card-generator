async function generateMessage(){ const name = document.getElementById("name").value.trim(); const age = document.getElementById("age").value.trim(); const hobby = document.getElementById("hobby").value.trim(); const style = document.getElementById("style").value; const result = document.getElementById("result"); const generateBtn = document.getElementById("generateBtn"); const optionsWrap = document.getElementById("optionsWrap"); const optionButtons = document.getElementById("optionButtons"); if(!name){ result.innerText = "Please enter at least a name 🎂"; optionsWrap.style.display = "none"; optionButtons.innerHTML = ""; return; } result.innerText = "Generating birthday magic... 🎉"; optionsWrap.style.display = "none"; optionButtons.innerHTML = ""; generatedMessages = []; selectedMessageIndex = 0; generateBtn.disabled = true; generateBtn.innerText = "Generating..."; try{ const response = await fetch(API_URL, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ name, age, hobby, style }) }); const rawText = await response.text(); let data = {}; try { data = rawText ? JSON.parse(rawText) : {}; } catch { data = { error: rawText || "Invalid server response" }; } if(!response.ok){ result.innerText = data.error || `Request failed with status ${response.status}`; return; } if(!Array.isArray(data.messages) || data.messages.length === 0){ result.innerText = "No messages generated."; return; } generatedMessages = data.messages; selectedMessageIndex = 0; result.innerText = generatedMessages[selectedMessageIndex]; renderOptionButtons(); optionsWrap.style.display = "block"; launchConfetti(); }catch(error){ console.error("Fetch failed:", error); result.innerText = `Could not reach AI service: ${error.message}`; }finally{ generateBtn.disabled = false; generateBtn.innerText = "Generate AI Message 🎉"; } }
